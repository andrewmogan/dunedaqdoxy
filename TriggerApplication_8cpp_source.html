85TriggerApplication::generate_modules(oksdbinterfaces::Configuration* confdb,
86 const std::string& dbfile,
87 const coredal::Session* session) const
88{
89 std::vector<const coredal::DaqModule*> modules;
90
91 auto ti_conf = get_trigger_inputs_handler();
92 auto ti_class = ti_conf->get_template_for();
93 std::string handler_name("");
94 // Process the queue rules looking for inputs to our trigger handler modules
95 const QueueDescriptor* ti_inputq_desc = nullptr;
96
97 for (auto rule : get_queue_rules()) {
98 auto destination_class = rule->get_destination_class();
99 auto data_type = rule->get_descriptor()->get_data_type();
100 if (destination_class == "ReadoutModule" || destination_class == ti_class) {
101 ti_inputq_desc = rule->get_descriptor();
102 }
103 }
104 // Process the network rules looking for the Fragment Aggregator and TP handler data reuest inputs
105 const NetworkConnectionDescriptor* req_net_desc = nullptr;
106 const NetworkConnectionDescriptor* tin_net_desc = nullptr;
107 const NetworkConnectionDescriptor* tout_net_desc = nullptr;
108 const NetworkConnectionDescriptor* tset_out_net_desc = nullptr;
109 for (auto rule : get_network_rules()) {
110 auto endpoint_class = rule->get_endpoint_class();
111 auto data_type = rule->get_descriptor()->get_data_type();
112
113 if (data_type == "DataRequest") {
114 req_net_desc = rule->get_descriptor();
115 }
116 else if (data_type == "TASet" || data_type == "TCSet"){
117 tset_out_net_desc = rule->get_descriptor();
118 }
119 else if (endpoint_class == "DataSubscriber") {
120 if (!tin_net_desc) {
121 tin_net_desc = rule->get_descriptor();
122 }
123 else if (rule->get_descriptor()->get_data_type() == tin_net_desc->get_data_type()) {
124 // For now endpoint_class of DataSubscriber for both input and output
125 // with the same data type is not possible.
126 throw (BadConf(ERS_HERE, "Have two network connections of the same data_type and the same endpoint_class"));
127 }
128 else if (tin_net_desc->get_data_type() == "TriggerActivity" &&
129 rule->get_descriptor()->get_data_type() == "TriggerCandidate") {
130 // For TA->TC
131 tout_net_desc = rule->get_descriptor();
132 handler_name = "tahandler";
133 }
134 else if (tin_net_desc->get_data_type() == "TriggerCandidate" &&
135 rule->get_descriptor()->get_data_type() == "TriggerActivity") {
136 // For TA->TC if we saved TC network connection as input first...
137 tout_net_desc = tin_net_desc;
138 tin_net_desc = rule->get_descriptor();
139 handler_name = "tahandler";
140 }
141 else {
142 throw (BadConf(ERS_HERE, "Unexpected input & output network connection descriptors provided"));
143 }
144 }
145 else if (data_type == "TriggerActivity" || data_type == "TriggerCandidate"){
146 tout_net_desc = rule->get_descriptor();
147 if (data_type == "TriggerActivity")
148 handler_name = "tphandler";
149 else
150 handler_name = "tahandler";
151 }
152 }
153
154 // Now create the Data Handler and its associated queue and network
155 // connections
156 oksdbinterfaces::ConfigObject input_queue_obj;
157 oksdbinterfaces::ConfigObject req_net_obj;
158 oksdbinterfaces::ConfigObject tin_net_obj;
159 oksdbinterfaces::ConfigObject tout_net_obj;
160 oksdbinterfaces::ConfigObject tset_out_net_obj;
161 auto handlerConf = get_trigger_inputs_handler();
162
163 if ( req_net_desc== nullptr) {
164 throw (BadConf(ERS_HERE, "No network descriptor given to receive request and send data was set"));
165 }
166 if ( tin_net_desc== nullptr) {
167 throw (BadConf(ERS_HERE, "No network descriptor given to receive trigger objects"));
168 }
169 if ( tout_net_desc== nullptr) {
170 throw (BadConf(ERS_HERE, "No network descriptor given to publish trigger objects"));
171 }
172 if (ti_inputq_desc == nullptr) {
173 throw (BadConf(ERS_HERE, "No data input queue descriptor given"));
174 }
175
176 std::string queue_uid(ti_inputq_desc->get_uid_base());
177 confdb->create(dbfile, "Queue", queue_uid, input_queue_obj);
178 input_queue_obj.set_by_val<std::string>("data_type", ti_inputq_desc->get_data_type());
179 input_queue_obj.set_by_val<std::string>("queue_type", ti_inputq_desc->get_queue_type());
180 input_queue_obj.set_by_val<uint32_t>("capacity", ti_inputq_desc->get_capacity());
181
182 auto req_service_obj = req_net_desc->get_associated_service()->config_object();
183 std::string req_net_uid(req_net_desc->get_uid_base()+UID());;
184 confdb->create(dbfile, "NetworkConnection", req_net_uid, req_net_obj);
185 req_net_obj.set_by_val<std::string>("connection_type", req_net_desc->get_connection_type());
186 req_net_obj.set_by_val<std::string>("data_type", req_net_desc->get_data_type());
187 req_net_obj.set_obj("associated_service", &req_service_obj);
188
189 auto tin_service_obj = tin_net_desc->get_associated_service()->config_object();
190 std::string t_in_stream_uid(tin_net_desc->get_uid_base()+".*");
191 confdb->create(dbfile, "NetworkConnection", t_in_stream_uid, tin_net_obj);
192 tin_net_obj.set_by_val<std::string>("data_type", tin_net_desc->get_data_type());
193 tin_net_obj.set_by_val<std::string>("connection_type", tin_net_desc->get_connection_type());
194 tin_net_obj.set_obj("associated_service", &tin_service_obj);
195
196 auto tout_service_obj = tout_net_desc->get_associated_service()->config_object();
197 std::string t_stream_uid(tout_net_desc->get_uid_base()+UID());
198 confdb->create(dbfile, "NetworkConnection", t_stream_uid, tout_net_obj);
199 tout_net_obj.set_by_val<std::string>("data_type", tout_net_desc->get_data_type());
200 tout_net_obj.set_by_val<std::string>("connection_type", tout_net_desc->get_connection_type());
201 tout_net_obj.set_obj("associated_service", &tout_service_obj);
202
203
204
205 if (tset_out_net_desc) {
206 auto tset_out_service_obj = tset_out_net_desc->get_associated_service()->config_object();
207 std::string tset_stream_uid(tset_out_net_desc->get_uid_base()+UID());
208 confdb->create(dbfile, "NetworkConnection", tset_stream_uid, tset_out_net_obj);
209 tset_out_net_obj.set_by_val<std::string>("data_type", tset_out_net_desc->get_data_type());
210 tset_out_net_obj.set_by_val<std::string>("connection_type", tset_out_net_desc->get_connection_type());
211 tset_out_net_obj.set_obj("associated_service", &tset_out_service_obj);
212 }
213
214 auto ti_conf_obj = ti_conf->config_object();
215 oksdbinterfaces::ConfigObject ti_obj;
216 if (get_source_id() == nullptr) {
217 throw(BadConf(ERS_HERE, "No source_id associated with this TriggerApplication!"));
218 }
219 uint32_t source_id = get_source_id()->get_sid();
220 std::string ti_uid(handler_name + "-" + std::to_string(source_id));
221 confdb->create(dbfile, ti_class, ti_uid, ti_obj);
222 ti_obj.set_by_val<uint32_t>("source_id", source_id);
223 ti_obj.set_obj("module_configuration", &ti_conf_obj);
224 ti_obj.set_objs("inputs", {&input_queue_obj, &req_net_obj});
225 if (tset_out_net_desc!= nullptr) {
226 ti_obj.set_objs("outputs", {&tout_net_obj, &tset_out_net_obj});
227 }
228 else {
229 ti_obj.set_objs("outputs", {&tout_net_obj});
230 }
231 // Add to our list of modules to return
232 modules.push_back(confdb->get<ReadoutModule>(ti_uid));
233
234
235 // Now create the DataSubscriber object
236 auto rdr_conf = get_data_subscriber();
237 if (rdr_conf == nullptr) {
238 throw (BadConf(ERS_HERE, "No DataReader configuration given"));
239 }
240
241 // Create a DataReader
242
243 std::string reader_uid("data-reader-"+UID());
244 std::string reader_class = rdr_conf->get_template_for();
245 oksdbinterfaces::ConfigObject reader_obj;
246 TLOG_DEBUG(7) << "creating OKS configuration object for Data subscriber class " << reader_class;
247 confdb->create(dbfile, reader_class, reader_uid, reader_obj);
248 reader_obj.set_objs("inputs", {&tin_net_obj} );
249 reader_obj.set_objs("outputs", {&input_queue_obj} );
250 reader_obj.set_obj("configuration", &rdr_conf->config_object());
251
252 modules.push_back(confdb->get<DataSubscriber>(reader_uid));
253
254 return modules;
255}
