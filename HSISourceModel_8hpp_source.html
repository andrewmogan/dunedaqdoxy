76 bool handle_payload(dfmessages::HSIEvent& data) // NOLINT(build/unsigned)
77 {
78 TLOG_DEBUG(1) << "Received HSIEvent with signal map " << data.signal_map << " and timestamp " << data.timestamp;

81 if (signal_info != m_signals.end()) {
82 // clang-format off
83 candidate.time_start = data.timestamp - signal_info->second.first; // time_start
84 candidate.time_end = data.timestamp + signal_info->second.second; // time_end,
85 // clang-format on
86 } else {
87 throw dunedaq::trigger::SignalTypeError(ERS_HERE, "HSI subscriber" , data.signal_map);
88 }
89
90 candidate.time_candidate = data.timestamp;
91 // throw away bits 31-16 of header, that's OK for now
92 candidate.detid = (uint)detdataformats::DetID::Subdetector::kDAQ ; // NOLINT(build/unsigned)
93 candidate.type = triggeralgs::TriggerCandidate::Type::kTiming;
94
95 candidate.algorithm = triggeralgs::TriggerCandidate::Algorithm::kHSIEventToTriggerCandidate;
96 candidate.inputs = {};
97
98 if (!m_data_sender->try_send(std::move(candidate), iomanager::Sender::s_no_block)) {
99 ++m_dropped_packets;
100 }
