46def make_moo_record(conf_dict,name,path='temptypes'):
47 fields = []
48 for pname,pvalue in conf_dict.items():
49 typename = None
50 if type(pvalue) == int:
51 typename = 'temptypes.temp_integer'
52 elif type(pvalue) == float:
53 typename = 'temptypes.temp_float'
54 elif type(pvalue) == str:
55 typename = 'temptypes.temp_string'
56 elif type(pvalue) == bool:
57 typename = 'temptypes.temp_boolean'
58 else:
59 raise Exception(f'Invalid config argument type: {type(pvalue)}')
60 fields.append(dict(name=pname,item=typename))
61 moo.otypes.make_type(schema='record', fields=fields, name=name, path=path)

79def get_trigger_bitwords(bitwords):
80 # create bitwords flags
81 final_bit_flags = []
82 for bitword in bitwords:
83 tmp_bits = []
84 for bit_name in bitword:
85 bit_value = trgdataformats.string_to_fragment_type_value(bit_name)
86 if bit_value == 0:
87 raise RuntimeError(f'One (or more) of provided MLT trigger bitwords is unknown! Please recheck the names...')
88 else:
89 tmp_bits.append(bit_value)
90 final_bit_flags.append(tmp_bits)
91
92 return final_bit_flags

95 prob = 0

254 # (PAR 2022-06-09) The max_latency_ms here should be
255 # kept smaller than the corresponding value in the
256 # downstream TAZipper. The reason is to avoid tardy
257 # sets at run stop, which are caused as follows:
258 #
259 # 1. The TPZipper receives its last input TPSets from
260 # multiple links. In general, the last time received
261 # from each link will be different (because the
262 # upstream readout senders don't all stop
263 # simultaneously). So there will be sets on one link
264 # that don't have time-matched sets on the other
265 # links. TPZipper sends these unmatched sets out after
266 # TPZipper's max_latency_ms milliseconds have passed,
267 # so these sets are delayed by
268 # "tpzipper.max_latency_ms"
269 #
270 # 2. Meanwhile, the TAZipper has also stopped
271 # receiving data from all but one of the readout units
272 # (which are stopped sequentially), and so is in a
273 # similar situation. Once tazipper.max_latency_ms has
274 # passed, it sends out the sets from the remaining
275 # live input, and "catches up" with the current time
276 #
277 # So, if tpzipper.max_latency_ms >
278 # tazipper.max_latency_ms, the TA inputs made from the
279 # delayed TPSets will certainly arrive at the TAZipper
280 # after it has caught up to the current time, and be
281 # tardy. If the tpzipper.max_latency_ms ==
282 # tazipper.max_latency_ms, then depending on scheduler
283 # delays etc, the delayed TPSets's TAs _may_ arrive at
284 # the TAZipper tardily. With tpzipper.max_latency_ms <
285 # tazipper.max_latency_ms, everything should be fine.
