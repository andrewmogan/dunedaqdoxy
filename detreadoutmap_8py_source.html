202 def _validate_streams(self, streams):
203 """Validates the list of stream entries"""
204
205 src_id_list = [s.src_id for s in streams]
206 geo_id_list = [s.geo_id for s in streams]
207
208 # Ensure source id uniqueness
209 dups_src_ids = [item for item, count in collections.Counter(src_id_list).items() if count > 1]
210 if len(dups_src_ids):
211 raise ValueError(f"Found duplicated source ids : {', '.join([str(i) for i in sorted(dups_src_ids)])}")
212
213 # Ensure geo id uniqueness

219 def _validate_rohosts(self, streams):
220 # Check RU consistency, i.e. only one kind type per readout host
221 host_label_map = {
222 'flx': 'host',
223 'eth': 'rx_host',
224 }
225
226 kind_m = defaultdict(set)
227 det_id_m = defaultdict(set)
228 for en in streams:
229 ro_host = getattr(en.parameters, host_label_map[en.kind])
230 kind_m[ro_host].add(en.kind)
231 det_id_m[ro_host].add(en.geo_id.det_id)
232
233 multi_kind_hosts = {k:v for k,v in kind_m.items() if len(v) > 1}
234 if multi_kind_hosts:
235 raise ValueError(f"Readout hosts with streams of different kinds are not supported. Found {multi_kind_hosts}")
236
237 multi_det_hosts = {k:v for k,v in det_id_m.items() if len(v) > 1}
238 if multi_det_hosts:
239 raise ValueError(f"Readout hosts with streams from different detectors are not supported. Found {multi_det_hosts}")

242 def _validate_eth(self, streams):
243 """
244 Apply rules:
245 - ip and mac pairing is strict (one-to-one)
246 - a mac can only belong to a single host
247 """
248
249
250 rx_mac_to_host = defaultdict(set)
251 rx_mac_to_ip = defaultdict(set)
252 rx_mac_to_iface = defaultdict(set)
253 rx_ip_to_mac = defaultdict(set)
254
255 tx_mac_to_host = defaultdict(set)
256 tx_mac_to_ip = defaultdict(set)
257 tx_ip_to_mac = defaultdict(set)
258
259 for s in streams:
260 if s.kind != 'eth':
261 continue
262
263 rx_mac_to_host[s.parameters.rx_mac].add(s.parameters.rx_host)
264 rx_mac_to_ip[s.parameters.rx_mac].add(s.parameters.rx_ip)
265 rx_mac_to_iface[s.parameters.rx_mac].add(s.parameters.rx_iface)
266 rx_ip_to_mac[s.parameters.rx_ip].add(s.parameters.rx_mac)
267
268 tx_mac_to_ip[s.parameters.tx_mac].add(s.parameters.tx_ip)
269 tx_ip_to_mac[s.parameters.tx_ip].add(s.parameters.tx_mac)
270 tx_mac_to_host[s.parameters.tx_mac].add(s.parameters.tx_host)
271
272
273 dup_rx_hosts = { k:v for k,v in rx_mac_to_host.items() if len(v) > 1}
274 dup_rx_macs = { k:v for k,v in rx_mac_to_ip.items() if len(v) > 1}
275 dup_rx_iface = { k:v for k,v in rx_mac_to_iface.items() if len(v) > 1}
276 dup_rx_ips = { k:v for k,v in rx_ip_to_mac.items() if len(v) > 1}
277
278 dup_tx_hosts = { k:v for k,v in tx_mac_to_host.items() if len(v) > 1}
279 dup_tx_macs = { k:v for k,v in tx_mac_to_ip.items() if len(v) > 1}
280 dup_tx_ips = { k:v for k,v in tx_ip_to_mac.items() if len(v) > 1}
281
282
283 errors = []
284 if dup_rx_hosts:
285 errors.append(f"Many rx hosts associated to the same rx mac {dup_rx_hosts}")
286 if dup_rx_macs:
287 errors.append(f"Many rx ips associated to the same rx mac {dup_rx_macs}")
288 if dup_rx_iface:
289 errors.append(f"Many rx interfaces associated to the same rx mac {dup_rx_iface}")
290 if dup_rx_ips:
291 errors.append(f"Many rx macs associated to the same rx ips {dup_rx_ips}")
292
293
294 if dup_tx_hosts:
295 errors.append(f"Many tx hosts associated to the same tx mac {dup_tx_hosts}")
296 if dup_tx_macs:
297 errors.append(f"Many tx macs associated to the same tx ips {dup_tx_macs}")
298 if dup_tx_ips:
299 errors.append(f"Many tx ips associated to the same tx mac {dup_tx_ips}")
300
301 # FIXME : Create a dedicated exception
302 if errors:
303 nl = r'\n'
304 raise RuntimeError(f"Ethernet streams validation failed: {nl.join(errors)}")
