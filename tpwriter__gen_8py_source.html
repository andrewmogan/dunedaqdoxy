28def get_tpwriter_app(
29 detector,
30 dataflow,
31 daq_common,
32 app_name,
33 file_label,
34 source_id,
35 SRC_GEO_ID_MAP,
36 DEBUG=False
37 ):
38 """Generate the json configuration for the readout and DF process"""
39
40 # Temp vars
41 OUTPUT_PATH = dataflow.tpset_output_path
42 APP_NAME = app_name
43 OPERATIONAL_ENVIRONMENT = detector.op_env
44 MAX_FILE_SIZE = dataflow.tpset_output_file_size
45 DATA_RATE_SLOWDOWN_FACTOR = daq_common.data_rate_slowdown_factor
46 CLOCK_SPEED_HZ = detector.clock_speed_hz
47 SOURCE_IDX=source_id
48 HOST=dataflow.host_tpw
49
50 ONE_SECOND_INTERVAL_TICKS = CLOCK_SPEED_HZ / DATA_RATE_SLOWDOWN_FACTOR
51
52 modules = []
53
54 modules += [DAQModule(name = 'tpswriter',
55 plugin = "TPStreamWriter",
56 conf = tpsw.ConfParams(tp_accumulation_interval_ticks=ONE_SECOND_INTERVAL_TICKS,
57 source_id=SOURCE_IDX,
58 data_store_parameters=hdf5ds.ConfParams(
59 name="tp_stream_writer",
60 operational_environment = OPERATIONAL_ENVIRONMENT,
61 directory_path = OUTPUT_PATH,
62 max_file_size_bytes = MAX_FILE_SIZE,
63 disable_unique_filename_suffix = False,
64 srcid_geoid_map=SRC_GEO_ID_MAP,
65 filename_parameters = hdf5ds.FileNameParams(
66 overall_prefix = file_label,
67 digits_for_run_number = 6,
68 file_index_prefix = "",
69 digits_for_file_index = 4,
70 writer_identifier = f"{APP_NAME}_tpswriter"),
71 file_layout_parameters = h5fl.FileLayoutParams(
72 record_name_prefix= "TimeSlice",
73 record_header_dataset_name = "TimeSliceHeader",
74 digits_for_record_number = 6,
75 digits_for_sequence_number = 0,
76 path_param_list = h5fl.PathParamList(
77 [h5fl.PathParams(detector_group_type="Detector_Readout",
78 detector_group_name="TPC",
79 element_name_prefix="Link")])))))]
80
81 mgraph=ModuleGraph(modules)
82
83 mgraph.add_endpoint(".*", f"tpswriter.tpset_source", "TPSet", Direction.IN, is_pubsub=True)
84
85 tpw_app = App(modulegraph=mgraph, host=HOST)
86
87 tpw_app.mounted_dirs += [{
88 'name': 'raw-data',
89 'physical_location':OUTPUT_PATH,
90 'in_pod_location':OUTPUT_PATH,
91 'read_only': False
92 }]
93
94 return tpw_app
