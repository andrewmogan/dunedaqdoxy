7class ModuleGraph:
8 """
9 A set of modules and connections between them.
10
11 modulegraph holds a dictionary of modules, with each module
12 knowing its (outgoing) connections to other modules in the
13 modulegraph.
14
15 Connections to other modulegraphs are represented by
16 `endpoints`. The endpoint's `external_name` is the "public" name
17 of the connection, which other modulegraphs should use. The
18 endpoint's `internal_name` specifies the particular module and
19 sink/source name which the endpoint is connected to, which may be
20 changed without affecting other applications.
21 """

23 def combine_queues(self, queues : List[Queue]):
24 output_queues = []
25
26 for q in queues:
27 match = False
28 for oq in output_queues:
29 if oq.name == q.name:
30 match = True
31 for push_mod in q.push_modules:
32 if push_mod not in oq.push_modules:
33 oq.push_modules.append(push_mod)
34 for pop_mod in q.pop_modules:
35 if pop_mod not in oq.pop_modules:
36 oq.pop_modules.append(pop_mod)
37 break
38 if not match:
39 output_queues.append(q)
40
41 return output_queues

60 def digraph(self):
61 deps = nx.DiGraph()
62 modules_set = set()
63
64 for module in self.modules:
65 if module.name in modules_set:
66 raise RuntimeError(f"Module {module.name} appears twice in the ModuleGraph")
67 deps.add_node(module.name, label=f"<<B>{module.name}</B><BR/>[{module.plugin}]>", shape='box', style='"rounded,filled"', fillcolor='lightskyblue1')
68 modules_set.add(module.name)
69
70 for queue in self.queues:
71 for pop_mod in queue.pop_modules:
72 for push_mod in queue.push_modules:
73 queue_start = push_mod.split(".")
74 if len(queue_start) != 2:
75 raise RuntimeError(f"Bad queue config!: {queue} output module must be specified as module.queue_name")
76 queue_end = pop_mod.split(".")
77 if len(queue_end) != 2:
78 raise RuntimeError(f"Bad queue config!: {queue} input module must be specified as module.queue_name")
79 deps.add_edge(queue_start[0], queue_end[0], label=queue.name)
80
81 for endpoint in self.endpoints:
82 if endpoint.internal_name is None:
83 continue
84 endpoint_internal_data = endpoint.internal_name.split(".")
85 if len(endpoint_internal_data) != 2:
86 raise RuntimeError(f'Bad endpoint!: {endpoint} internal_endpoint must be specified as module.queue_name')
87 to_module = endpoint_internal_data[0]
88 if to_module in modules_set:
89 if endpoint.direction == Direction.IN:
90 deps.add_edge(endpoint.external_name, to_module, label=endpoint_internal_data[1], color='red', style='dashed')
91 deps.nodes[endpoint.external_name]['color'] = 'red'
92 deps.nodes[endpoint.external_name]['shape'] = 'box'
93 else:
94 deps.add_edge(to_module, endpoint.external_name, label=endpoint_internal_data[1], color='blue', style='dashed')
95 deps.nodes[endpoint.external_name]['color'] = 'blue'
96 deps.nodes[endpoint.external_name]['shape'] = 'box'
97 else:
98 raise RuntimeError(f"Bad endpoint {endpoint}: internal connection which doesn't connect to any module! Available modules: {modules_set}")
99
100 # finally the fragment producers
101 for producer in self.fragment_producers.values():
102 producer_request_in = producer.requests_in.split(".")
103 if len(producer_request_in) != 2:
104 raise RuntimeError(f"Bad fragment producer {producer}: request_in must be specified as module.queue_name")
105
106 if producer_request_in[0] in modules_set:
107 deps.add_edge("TriggerRecordBuilder", producer_request_in[0], label='requests', color='gold', style='dashed')
108 deps.nodes["TriggerRecordBuilder"]['color'] = 'gold'
109 deps.nodes["TriggerRecordBuilder"]['style'] = 'filled, dashed'
110 deps.nodes["TriggerRecordBuilder"]['shape'] = 'oval'
111 deps.nodes["TriggerRecordBuilder"]['label'] = 'from TRB'
112 else:
113 raise RuntimeError(f"Bad FragmentProducer {producer}: request_in doesn't connect to any module! Available modules: {modules_set}")
114
115 producer_frag_out = producer.fragments_out.split(".")
116 if len(producer_frag_out) != 2:
117 raise RuntimeError(f"Bad fragment producer {producer}: fragments_out must be specified as module.queue_name")
118
119 if producer_frag_out[0] in modules_set:
120 deps.add_edge(producer_frag_out[0], "FragmentReceiver", label='fragments', color='orange', style='dashed, filled')
121 deps.nodes["FragmentReceiver"]['color'] = 'orange'
122 deps.nodes["FragmentReceiver"]['style'] = 'filled, dashed'
123 deps.nodes["FragmentReceiver"]['shape'] = 'oval'
124 deps.nodes["FragmentReceiver"]['label'] = 'to FR'
125
126
127 else:
128 raise RuntimeError(f"Bad FragmentProducer {producer}: fragments_out doesn't connect to any module! Available modules: {modules_set}")

145 def reset_module_conf(self, name, new_conf):
146 """Replace the configuration object of the module `name` with the new object `conf`"""
147 # It would be nice to just modify the `conf` attribute of the
148 # DAQModule object directly, but moo-derived objects work in a funny
149 # way (returning a copy of the attribute, not returning a
150 # reference to it), which means we have to copy and replace the
151 # whole module
152 for i,mod in enumerate(self.modules):
153 if mod.name == name:
154 old_module = self.modules[i]
155 new_module = DAQModule(name=name,
156 plugin=old_module.plugin,
157 conf=new_conf)
158 self.modules[i] = new_module
159 return
160 raise RuntimeError(f'Module {name} not found!')

194 def connect_modules(self, push_addr:str, pop_addr:str, data_type:str, queue_name:str = "", size_hint:int = 10, toposort = True):
195 queue_start = push_addr.split(".")
196 queue_end = pop_addr.split(".")
197 if len(queue_start) < 2 or queue_start[0] not in self.module_names():
198 raise RuntimeError(f"connect_modules called with invalid parameters. push_addr ({push_addr}) must be of form <module>.<internal name>, and the module must already be in the module graph!")
199
200 if len(queue_end) < 2 or queue_end[0] not in self.module_names():
201 raise RuntimeError(f"connect_modules called with invalid parameters. pop_addr ({pop_addr}) must be of form <module>.<internal name>, and the module must already be in the module graph!")
202
203 if queue_name == "":
204 self.queues.append(Queue(push_addr, pop_addr, data_type, push_addr + "_to_" + pop_addr, size_hint, toposort))
205 else:
206 existing_queue = False
207 for queue in self.queues:
208 if queue.name == queue_name:
209 queue.add_module_link(push_addr, pop_addr)
210 existing_queue = True
211 if not existing_queue:
