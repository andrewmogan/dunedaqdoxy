32def get_dataflow_app(
33 df_config,
34 dataflow,
35 detector,
36 HOSTIDX=0,
37 APP_NAME="dataflow0",
38 FILE_LABEL="swtest",
39 MAX_EXPECTED_TR_SEQUENCES=1,
40 TRB_TIMEOUT=200,
41 SRC_GEO_ID_MAP='',
42 DEBUG=False
43 ):
44
45 """Generate the json configuration for the readout and DF process"""
46
47 OUTPUT_PATHS = df_config.output_paths
48 OPERATIONAL_ENVIRONMENT = detector.op_env
49 DATA_STORE_MODE=df_config.data_store_mode
50 MAX_FILE_SIZE = df_config.max_file_size
51 MAX_TRIGGER_RECORD_WINDOW = df_config.max_trigger_record_window
52 TOKEN_COUNT = dataflow.token_count
53 HOST=df_config.host_df
54
55
56 modules = []
57 queues = []
58
59 if DEBUG: print(f"dataflow{HOSTIDX}: Adding TRB instance")
60 modules += [DAQModule(name = 'trb',
61 plugin = 'TriggerRecordBuilder',
62 conf = trb.ConfParams(general_queue_timeout=QUEUE_POP_WAIT_MS,
63 max_time_window=MAX_TRIGGER_RECORD_WINDOW,
64 source_id = HOSTIDX,
65 trigger_record_timeout_ms=TRB_TIMEOUT))]
66
67 queue_size_based_on_number_of_sequences = max(10, int(MAX_EXPECTED_TR_SEQUENCES * TOKEN_COUNT * 1.1))
68 for i in range(len(OUTPUT_PATHS)):
69 if DEBUG: print(f"dataflow{HOSTIDX}: Adding datawriter{i} instance")
70 modules += [DAQModule(name = f'datawriter_{i}',
71 plugin = 'DataWriter',
72 conf = dw.ConfParams(decision_connection=f"trigger_decision_{HOSTIDX}",
73 data_store_parameters=hdf5ds.ConfParams(
74 name="data_store",
75 operational_environment = OPERATIONAL_ENVIRONMENT,
76 mode = DATA_STORE_MODE,
77 directory_path = OUTPUT_PATHS[i],
78 max_file_size_bytes = MAX_FILE_SIZE,
79 disable_unique_filename_suffix = False,
80 srcid_geoid_map=SRC_GEO_ID_MAP,
81 filename_parameters = hdf5ds.FileNameParams(
82 overall_prefix = FILE_LABEL,
83 digits_for_run_number = 6,
84 file_index_prefix = "",
85 digits_for_file_index = 4,
86 writer_identifier = f"{APP_NAME}_datawriter_{i}"),
87 file_layout_parameters = h5fl.FileLayoutParams(
88 record_name_prefix= "TriggerRecord",
89 digits_for_record_number = 5,
90 path_param_list = h5fl.PathParamList(
91 [h5fl.PathParams(detector_group_type="Detector_Readout",
92 detector_group_name="TPC",
93 element_name_prefix="Link"),
94 h5fl.PathParams(detector_group_type="Detector_Readout",
95 detector_group_name="PDS"),
96 h5fl.PathParams(detector_group_type="Detector_Readout",
97 detector_group_name="NDLArTPC"),
98 h5fl.PathParams(detector_group_type="Detector_Readout",
99 detector_group_name="NDLArPDS"),
100 h5fl.PathParams(detector_group_type="Trigger",
101 detector_group_name="DataSelection",
102 digits_for_element_number=5),
103 h5fl.PathParams(detector_group_type="HW_Signals_Interface",
104 detector_group_name="HSI")
105 ])))))]
106
107 queues += [Queue("trb.trigger_record_output",
108 f"datawriter_{i}.trigger_record_input",
109 "TriggerRecord",
110 "trigger_records",
111 queue_size_based_on_number_of_sequences)
112 ]
113
114 mgraph=ModuleGraph(modules, queues = queues)
115
116 mgraph.add_endpoint(f"trigger_decision_{HOSTIDX}", "trb.trigger_decision_input", "TriggerDecision", Direction.IN)
117
118 for i in range(len(OUTPUT_PATHS)):
119 mgraph.add_endpoint("triginh", f"datawriter_{i}.token_output", "TriggerDecisionToken", Direction.OUT, toposort=True)
120
121 df_app = App(modulegraph=mgraph, host=HOST)
122
123
124 df_app.mounted_dirs += [{
125 'name': f'raw-data-{i}',
126 'physical_location': opath,
127 'in_pod_location': opath,
128 'read_only': False,
129 } for i,opath in enumerate(set(OUTPUT_PATHS))]
130
131 return df_app
