39
40

46 TLOG(1) << "Emitting a trigger for candidate Michel event.";
47 output_ta.push_back(construct_ta());
48 m_current_window.reset(input_tp);
49 } // Kinks
50 } // Bragg peak
51
52 }
53

112std::vector<TriggerPrimitive>
113TriggerActivityMakerMichelElectron::longest_activity() const
114{
115 // This function attempts to return a vector of hits that correspond to the longest
116 // piece of activity in the current window. The logic follows that from the HMA
117 // check_adjacency() function and further details can be found there.
118 std::vector<TriggerPrimitive> trackHits;
119 std::vector<TriggerPrimitive> finalHits; // The vector of track hits, which we return
120
121 uint16_t adj = 1; // Initialise adjacency, 1 for the first wire.
122 uint16_t max = 0;
123 unsigned int channel = 0; // Current channel ID
124 unsigned int next_channel = 0; // Next channel ID
125 unsigned int next = 0; // The next position in the hit channels vector
126 unsigned int tol_count = 0; // Tolerance count, should not pass adj_tolerance
127
128 // Generate a channelID ordered list of hit channels for this window
129 std::vector<TriggerPrimitive> hitList;
130 for (auto tp : m_current_window.inputs) {
131 hitList.push_back(tp);
132 }
133 std::sort(hitList.begin(), hitList.end(), [](TriggerPrimitive a, TriggerPrimitive b)
134 { return a.channel < b.channel; });
135
136 // ADAJACENCY LOGIC ====================================================================
137 // =====================================================================================
138 // Adjcancency Tolerance = Number of times prepared to skip missed hits before resetting
139 // the adjacency count. This accounts for things like dead channels / missed TPs. The
140 // maximum gap is 4 which comes from tuning on December 2021 coldbox data, and June 2022
141 // coldbox runs.
142 for (int i = 0; i < hitList.size(); ++i) {
143
144 next = (i + 1) % hitList.size(); // Loops back when outside of channel list range
145 channel = hitList.at(i).channel;
146 next_channel = hitList.at(next).channel; // Next channel with a hit
147
148 if (trackHits.size() == 0 ){ trackHits.push_back(hitList.at(i)); }
149
150 // End of vector condition.
151 if (next_channel == 0) { next_channel = channel - 1; }
152
153 // Skip same channel hits for adjacency counting, but add to the track!
154 if (next_channel == channel) {
155 trackHits.push_back(hitList.at(next));
156 continue; }
157
158 // If next hit is on next channel, increment the adjacency count.
159 else if (next_channel == channel + 1){
160 trackHits.push_back(hitList.at(next));
161 ++adj; }
162
163 // If next channel is not on the next hit, but the 'second next', increase adjacency
164 // but also tally up with the tolerance counter.
165 else if (((next_channel == channel + 2) || (next_channel == channel + 3) ||
166 (next_channel == channel + 4) || (next_channel == channel + 5))
167 && (tol_count < m_adj_tolerance)) {
168 trackHits.push_back(hitList.at(next));
169 ++adj;
170 for (int i = 0 ; i < next_channel-channel ; ++i){ ++tol_count; }
171 }
172
173 // If next hit isn't within reach, end the adjacency count and check for a new max.
174 // Reset variables for next iteration.
175 else {
176 if (adj > max) {
177 max = adj;
178 finalHits.clear(); // Clear previous track
179 for (auto h : trackHits){
180 finalHits.push_back(h);
181 }
182 }
183 adj = 1;
184 tol_count = 0;
185 trackHits.clear();
186 }
187 }
188
189 return finalHits;
190}

193// Function that tries to identify a Bragg peak via a running
194// mean average of the ADC values. We use the running mean as it's less susceptible to
195// spikes of activity that might trick the algorithm. We establish a baseline, then
196// count up clusters of charge deposition above that baseline. If the largest is at
197// one of the ends of that collection, signal a potential Bragg peak.
198bool
199TriggerActivityMakerMichelElectron::check_bragg_peak(std::vector<TriggerPrimitive> trackHits)
200{
201 bool bragg = false;
202 std::vector<float> adc_means_list;
203 uint16_t convolve_value = 6;
204
205 // Loop over hits that correspond to high adjacency activity
206 for (uint16_t i = 0; i < trackHits.size(); ++i){
207 float adc_sum = 0;
208 float adc_mean = 0;
209
210 // Calculate running ADC mean of this track
211 for (uint16_t j = i; j < i+convolve_value; ++j){
212 int hit = (j) % trackHits.size();
213 adc_sum += trackHits.at(hit).adc_integral;
214 }
215
216 adc_mean = adc_sum / convolve_value;
217 adc_means_list.push_back(adc_mean);
218 adc_sum = 0;
219 }
220
221 // We now have a list of convolved adc means.
222 float ped = std::accumulate(adc_means_list.begin(), adc_means_list.end(), 0.0) / adc_means_list.size();
223 float charge = 0;
224 std::vector<float> charge_dumps;
225
226 // Now go through the list, picking up clusters of charge above the baseline/ped
227 for (auto a : adc_means_list){
228 if (a > ped){
229 charge += a;
230 }
231 else if( a < ped && charge !=0 ){
232 charge_dumps.push_back(charge);
233 charge = 0;
234 }
235 }
236
237 // If the maximum of that list of charge dumps is near(at?) either end of it
238 float max_charge = *max_element(charge_dumps.begin(), charge_dumps.end());
239 if(max_charge == charge_dumps.front() || max_charge == charge_dumps.back()){ bragg=true; }
240
241 return bragg;
242 }

244bool
245TriggerActivityMakerMichelElectron::check_kinks(std::vector<TriggerPrimitive> finalHits)
246{
247 bool kinks = false; // We actually required two kinks in the coldbox, the michel kink and the wes kink
248 std::vector<float> runningGradient;
249 std::vector<float> runningMeanGradient;
250
251 // Choice to be made here. Do we want to scane in collection (z) or time (x) direction when calculating gradient between hits. I
252 // would say if we have already made the request to pass a track of length specific threshold which is longer than the drift
253 // direction for the coldbox, it makes sense to scan across channels a little more.
254 std::sort(finalHits.begin(), finalHits.end(), [](TriggerPrimitive a, TriggerPrimitive b) { return a.channel < b.channel; });
255
256 // Populate the runningGradient with the track hits. Do this between ith and i+kth TPs, to small scale fluctuations of the track
257 // Yet k should be kept small, so that enough gradient information is preserved at the end of the track to identify kinks
258 for (int i=0 ; i < finalHits.size()-2; i++){
259
260 // Skip same channel hits or if the start times are the same - no div by zero!
261 if (finalHits.at(i+2).channel == finalHits.at(i).channel || (finalHits.at(i+2).time_start == finalHits.at(i).time_start) ) { continue; }
262
263 // Check that the next TP is closeby; enough in space and time directions so as to avoid obtaining a gradient value from
264 // same channel hits at large time difference or vice versa due to kink topology or showers. Clearly we shouldn't be very far in
265 // channel number, but since we might later try to do this check in the time direction, leave the condition in.
266 int diff = finalHits.at(i+2).time_start - finalHits.at(i).time_start;
267 if((std::abs(diff) > 1000) || ((std::abs(finalHits.at(i+2).channel - finalHits.at(i).channel) > 6))) { continue; }
268
269 // Gradient is just change in z (collection) over change in x (drift). x is admitedly roughly converted from
270 // hit start time, but I don't think diffusion effects are a huge concern over 20cm. Using mm for readability/visualisation
271 float dz = (finalHits.at(i+2).channel - finalHits.at(i).channel)*4.67; // Change in collection wire z to separation in mm
272 long long int dt = finalHits.at(i+2).time_start - finalHits.at(i).time_start;
273 float dx = dt*0.028; // Change time to separation in x mm
274 float g = dz/dx;
275
276 runningGradient.push_back(g);
277 }
278
279 // Require a decent length of the gradients vector. Otherwise some adjacent events are showers and the conditions above mean
280 // we don't get enough entries. In essence, this provides some confidence that it's track-like rather than shower-like. Which
281 // is what we want for a Michel event.
282 if ( runningGradient.size() > 10 ){
283
284 // Now lets take a running mean of the gradients between TPs, less susceptible to wild changes due to deltas/etc
285 for(int g=0 ; g < runningGradient.size()-1 ; g++){
286 float gsum = runningGradient.at(g) + runningGradient.at(g+1);
287 runningMeanGradient.push_back(gsum/2);
288 }
289
290 // We have a list of gradients, now just demand that the two ends have gradients that differ significantly
291 // from the mean gradient of the activity at both ends. This aims to pick out wesKinks and michelKinks
292 if(runningMeanGradient.size() > 10 ){
293
294 float mean = (std::abs(std::accumulate(runningMeanGradient.begin(), runningMeanGradient.end(), 0.0)))/(runningMeanGradient.size());
295
296 // If you're testing on simulation or december data, you won't see the wes kink, so use an || instead of &&
297 if((std::abs(runningMeanGradient.front()) + mean > 2.5*mean) || ((std::abs(runningMeanGradient.back() + mean)) > 2.5*mean)){ kinks=true; }
298 }
299 }
300
301 return kinks;
302}
