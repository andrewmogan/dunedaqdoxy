64def _strict_recursive_update(dico1, dico2):

82def parse_json(filename, schemed_object):
83 console.log(f"Parsing config json file {filename}")
84
85 filepath = Path(filename)
86 # basepath = filepath.parent
87
88 # First pass, load the main json file
89 with open(filepath, 'r') as f:
90 try:
91 new_parameters = json.load(f)
92 except Exception as e:
93 raise RuntimeError(f"Couldn't parse {filepath}, error: {str(e)}")
94
95 # second pass, look for references
96 subkeys = [ k for k,v in schemed_object.pod().items() if isinstance(v,dict) ]
97 for k in new_parameters:
98 # look for keys that are associated to dicts in the schemed_obj but here are strings
99 v = new_parameters[k]
100
101 if isinstance(v,str) and k in subkeys:
102
103 # It's a string! It's a reference! Try loading it
104 subfile_path = Path(os.path.expandvars(v)).expanduser()
105 if not subfile_path.is_absolute():
106 subfile_path = filepath.parent / subfile_path
107 if not subfile_path.exists():
108 raise RuntimeError(f'Cannot find the file {v} ({subfile_path})')
109
110 console.log(f"Detected subconfiguration for {k} {v} - loading {subfile_path}")
111 with open(subfile_path, 'r') as f:
112 try:
113 new_subpars = json.load(f)
114 except Exception as e:
115 raise RuntimeError(f"Couldn't parse {subfile_path}, error: {str(e)}")
116 new_parameters[k] = new_subpars
117
118 elif '<defaults>' in v:
119 cname = k
120 pars = v['<defaults>']
121 scname = pars['config_name']
122 scfile = pars['config_file'] if 'config_file' in pars else f'{cname}_configs.json'
123 scbase = pars['config_base'] if 'config_base' in pars else 'common'
124
125
126 scfile = Path(os.path.expandvars(scfile)).expanduser()
127 if not scfile.is_absolute():
128 scfile = filepath.parent / scfile
129
130 if not scfile.exists():
131 raise RuntimeError(f'Cannot find the file {v} ({scfile})')
132
133 scset = ConfigSet(scfile,scbase)
134 new_parameters[k] = scset.get(scname)
135
136 try:
137 # Validate the heck out of this but that doesn't change the object itself (ARG)
138 _strict_recursive_update(schemed_object.pod(), new_parameters)
139 # now its validated, update the object with moo
140 schemed_object.update(new_parameters)
141 except Exception as e:
142 raise RuntimeError(f'Couldn\'t update the object {schemed_object} with the file {filename},\nError: {e}')
143
144 return schemed_object

166def helptree(ost, prefix=''):
167 if 'doc' in ost.keys():
168 output = f"{prefix}{ost['name']}: {ost['doc']}"
169 else:
170 output = f"{prefix}{ost['name']}:"
171 for field in ost['fields']:
172
173 if type(field['default']) is dict:
174 output += "\n" + helptree(field["default"], prefix + " ")
175 else:
176 docstr = ""
177 if 'doc' in field.keys():
178 docstr = f": {field['doc']}"
179 output += f"\n{prefix} {field['name']} (Default: {field['default']}){docstr}"
