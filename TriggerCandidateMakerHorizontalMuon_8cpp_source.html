146TriggerCandidate
147TriggerCandidateMakerHorizontalMuon::construct_tc() const
148{
149 TriggerActivity latest_ta_in_window = m_current_window.inputs.back();
150
151 TriggerCandidate tc;
152 tc.time_start = m_current_window.time_start - m_readout_window_ticks_before;
153 tc.time_end = m_current_window.time_start + m_readout_window_ticks_after;
154 //tc.time_end = latest_ta_in_window.inputs.back().time_start + latest_ta_in_window.inputs.back().time_over_threshold;
155 tc.time_candidate = m_current_window.time_start;
156 tc.detid = latest_ta_in_window.detid;
157 tc.type = TriggerCandidate::Type::kHorizontalMuon;
158 tc.algorithm = TriggerCandidate::Algorithm::kHorizontalMuon;
159
160 // Take the list of triggeralgs::TriggerActivity in the current
161 // window and convert them (implicitly) to detdataformats'
162 // TriggerActivityData, which is the base class of TriggerActivity
163 for (auto& ta : m_current_window.inputs) {
164 tc.inputs.push_back(ta);
165 }
166
167 return tc;
168}
