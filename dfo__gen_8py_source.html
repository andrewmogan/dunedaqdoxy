23def make_moo_record(conf_dict,name,path='temptypes'):
24 fields = []
25 for pname,pvalue in conf_dict.items():
26 typename = None
27 if type(pvalue) == int:
28 typename = 'temptypes.temp_integer'
29 elif type(pvalue) == float:
30 typename = 'temptypes.temp_float'
31 elif type(pvalue) == str:
32 typename = 'temptypes.temp_string'
33 else:
34 raise Exception(f'Invalid config argument type: {type(pvalue)}')
35 fields.append(dict(name=pname,item=typename))
36 moo.otypes.make_type(schema='record', fields=fields, name=name, path=path)
