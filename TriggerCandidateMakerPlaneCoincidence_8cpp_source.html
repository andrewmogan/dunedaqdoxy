26 // The first time operator is called, reset window object.
27 if (m_current_window.is_empty()) {
28 m_current_window.reset(activity);
29 m_activity_count++;

31 // If the request has been made to not trigger on number of channels or
32 // total adc, simply construct a trigger candidate from any single activity.

34
35 // add_window_to_record(m_current_window);

48 // FIX ME: Only want to call this if running in debug mode.
49 // add_window_to_record(m_current_window);

51 // If the difference between the current TA's start time and the start of the window
52 // is less than the specified window size, add the TA to the window.
53 if ((activity.time_start - m_current_window.time_start) < m_window_length) {
54 // TLOG_DEBUG(TRACE_NAME) << "TAWindow not yet complete, adding the activity to the window.";
55 m_current_window.add(activity);

57 // If the addition of the current TA to the window would make it longer
58 // than the specified window length, don't add it but check whether the sum of all adc in
59 // the existing window is above the specified threshold. If it is, and we are triggering on ADC,
60 // make a TA and start a fresh window with the current TP.

70 // than the specified window length, don't add it but check whether the number of hit channels in
71 // the existing window is above the specified threshold. If it is, and we are triggering on channels,
72 // make a TC and start a fresh window with the current TA.
73 else if (m_current_window.n_channels_hit() > m_n_channels_threshold && m_trigger_on_n_channels) {
74 tc_number++;

81 // TLOG_DEBUG(TRACE_NAME) << "TAWindow is at required length but specified threshold not met, shifting window along.";
82 m_current_window.move(activity, m_window_length);

114TriggerCandidate
115TriggerCandidateMakerPlaneCoincidence::construct_tc() const
116{
117 TriggerActivity latest_ta_in_window = m_current_window.inputs.back();
118
119 TriggerCandidate tc;
120 tc.time_start = m_current_window.time_start - m_readout_window_ticks_before;
121 tc.time_end =
122 latest_ta_in_window.inputs.back().time_start + latest_ta_in_window.inputs.back().time_over_threshold + m_readout_window_ticks_after;
123 tc.time_candidate = m_current_window.time_start;
124 tc.detid = latest_ta_in_window.detid;
125 tc.type = TriggerCandidate::Type::kPlaneCoincidence;
126 tc.algorithm = TriggerCandidate::Algorithm::kPlaneCoincidence;
127
128 // Take the list of triggeralgs::TriggerActivity in the current
129 // window and convert them (implicitly) to detdataformats'
130 // TriggerActivityData, which is the base class of TriggerActivity
131 for (auto& ta : m_current_window.inputs) {
132 tc.inputs.push_back(ta);
133 }
134
135 return tc;
136}
