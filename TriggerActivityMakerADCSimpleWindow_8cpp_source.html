18void
19TriggerActivityMakerADCSimpleWindow::operator()(const TriggerPrimitive& input_tp, std::vector<TriggerActivity>& output_ta)
20{
21
22 // The first time operator is called, reset
23 // window object.
24 if(m_current_window.is_empty()){
25 m_current_window.reset(input_tp);
26 m_primitive_count++;
27 return;
28 }
29
30 // If the difference between the current TP's start time and the start of the window
31 // is less than the specified window size, add the TP to the window.
32 if((input_tp.time_start - m_current_window.time_start) < m_window_length){
33 //TLOG_DEBUG(TRACE_NAME) << "Window not yet complete, adding the input_tp to the window.";
34 m_current_window.add(input_tp);
35 }
36 // If the addition of the current TP to the window would make it longer
37 // than the specified window length, don't add it but check whether the sum of all adc in
38 // the existing window is above the specified threshold. If it is, make a TA and start
39 // a fresh window with the current TP.
40 else if(m_current_window.adc_integral > m_adc_threshold){
41 //TLOG_DEBUG(TRACE_NAME) << "ADC integral in window is greater than specified threshold.";
42 output_ta.push_back(construct_ta());
43 //TLOG_DEBUG(TRACE_NAME) << "Resetting window with input_tp.";
44 m_current_window.reset(input_tp);
45 }
46 // If it is not, move the window along.
47 else{
48 //TLOG_DEBUG(TRACE_NAME) << "Window is at required length but adc threshold not met, shifting window along.";
49 m_current_window.move(input_tp, m_window_length);
50 }
51
52 //TLOG_DEBUG(TRACE_NAME) << m_current_window;
53
54 m_primitive_count++;
55
56 return;
57}
